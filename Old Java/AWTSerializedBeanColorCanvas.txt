import java.awt.*;
import java.awt.event.*;
import java.serializable.*;

public class AWTCanvas extends Canvas implements Serializable{
	public boolean rectangular;
	transient Color color;	

	AWTCanvas(){
		rectangular = false;
		change();
		setSize(200, 150);
		addMouseListener(new MouseAdapter(){
			public void onMousePressed(MouseEvent me){
				change();
			}
		}
	}


	public boolean getRectangular(){
		return rectangular;
	}


	public void setRectangular(boolean flagging){
		this.rectangular = flagging;
		change();
	}

	private void change(){
		color = randomColor();
		repaint();
	}

	private Color randomColor(){
		int r = (255)*(int)(Math.random());
		int g = (255)*(int)(Math.random());
		int b = (255)*(int)(Math.random());
		
		return new Color(r, g, b);
	}

	public void paint(Graphics g){
		g.setColor(color);
		
		Dimension d = getSize();
		int h = d.height;
		int w = d.width;
	
		if(rectangular){
			g.fillRect(0, 0, w, h);
		}
		else{
			g.fillOval(0, 0, w, h);
		}
	}
}
